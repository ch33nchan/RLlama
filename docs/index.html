BASIC REWARD COMPOSITION

BEGINNER CORE
BASIC_EXAMPLE.PY 
from rllama import RewardEngine

# Initialize with configuration
engine = RewardEngine("config.yaml", verbose=True)

# Compute reward
context = {
    "response": "This demonstrates RLlama's capabilities.",
    "query": "Tell me about RLlama",
    "metadata": {"model": "gpt-4", "temperature": 0.7}
}

reward = engine.compute_and_log(context)
print(f"Final reward: {reward:.4f}")

# Get component breakdown
component_rewards = engine.composer.calculate(context)
for component, value in component_rewards.items():
    print(f"{component}: {value:.4f}")
