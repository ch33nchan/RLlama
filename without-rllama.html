Integration with RL Frameworks

RLlama is designed to integrate seamlessly with popular reinforcement learning frameworks. This guide covers integration with:

  • OpenAI Gym environments
  • Stable Baselines3
  • Custom RL environments
  • Deep learning frameworks (e.g., PyTorch)
  • Logging & Monitoring

OpenAI Gym Integration

RLlama provides a wrapper for Gym environments that intercepts and modifies rewards:

from rllama import RewardEngine
from rllama.integration import GymWrapper
from rllama.rewards.components import ProgressReward, SparseSuccessReward
import gym

# Create a standard Gym environment
env = gym.make('CartPole-v1')

# Create and configure a reward engine
engine = RewardEngine()
engine.add_component(ProgressReward(
    goal_pos=0.0,
    goal_threshold=0.05,
    state_key=0
))
engine.add_component(SparseSuccessReward(
    reward_value=10.0
))

# Wrap the environment
wrapped_env = GymWrapper(engine).wrap(env)

# Use as usual
obs, info = wrapped_env.reset()
for _ in range(1000):
    action = wrapped_env.action_space.sample()
    obs, reward, terminated, truncated, info = wrapped_env.step(action)
    if terminated or truncated:
        obs, info = wrapped_env.reset()

Custom Context Builder

Define a function to customize the context passed to reward components:

def custom_context_builder(state, action, next_state, reward, done, info):
    return {
        "state": state,
        "action": action,
        "next_state": next_state,
        "original_reward": reward,
        "done": done,
        "info": info,
        "position": next_state[0],
        "velocity": next_state[1],
        "distance_to_goal": abs(next_state[0])
    }

wrapped_env = GymWrapper(
    engine,
    context_builder=custom_context_builder,
    mode="add"   # "replace", "add", or "observe"
).wrap(env)

Stable Baselines3 Integration

from rllama.integration import StableBaselinesWrapper
from stable_baselines3 import PPO

# Wrap your env as above
wrapped_env = GymWrapper(engine).wrap(env)

# Use SB3 wrapper for easy model creation
sb3_wrapper = StableBaselinesWrapper(engine)
model = sb3_wrapper.create_model(
    algorithm="PPO",
    env=wrapped_env,
    policy="MlpPolicy",
    learning_rate=0.0003,
    n_steps=2048,
    batch_size=64,
    n_epochs=10,
    gamma=0.99,
    verbose=1
)

# Train and save
model.learn(total_timesteps=100_000, log_interval=100, log_dir="./sb3_logs")
model.save("ppo_cartpole_with_rllama")

Custom RL Environment Integration

Integrate RLlama directly in your own environment class:

class MyCustomEnv:
    def __init__(self):
        self.state = self.reset()
        self.engine = RewardEngine()
        # add components...
    
    def reset(self):
        self.state = initial_state
        return self.state
    
    def step(self, action):
        prev = self.state
        self.state = self.update_state(action)
        done = self.is_done()
        context = {
            "state": prev,
            "action": action,
            "next_state": self.state,
            "done": done
        }
        reward = self.engine.compute(context)
        return self.state, reward, done, {}

Deep Learning Frameworks (PyTorch)

import torch, torch.nn as nn
from rllama.models import MLPRewardModel

# Define a neural reward model
model = MLPRewardModel(input_dim=state_dim+action_dim, hidden_dims=[128,64], output_dim=1)
engine.add_component(NeuralReward(model=model))

# Train the reward model
opt = torch.optim.Adam(model.parameters(), lr=1e-3)
loss_fn = nn.MSELoss()

for epoch in range(num_epochs):
    states, actions, targets = get_batch()
    inputs = torch.cat([states, actions], dim=1)
    preds = model(inputs).squeeze()
    loss = loss_fn(preds, targets)
    opt.zero_grad()
    loss.backward()
    opt.step()
